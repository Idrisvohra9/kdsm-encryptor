## ‚úÖ 1. Create a `chatRooms` Document

```ts
import { ID, databases } from "@/lib/appwrite"; // assuming alias path setup

export async function createChatRoom({
  name,
  creatorId,
  roomKeyHash = "",
  members = [],
  retention = "3days",
  autoDecrypt = false,
}) {
  return await databases.createDocument(
    "yourDatabaseId",
    "chatRooms",
    ID.unique(),
    {
      name,
      creatorId,
      roomKeyHash,
      members,
      retention,
      autoDecrypt,
      createdAt: new Date().toISOString(),
    }
  );
}
```
